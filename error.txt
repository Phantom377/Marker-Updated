import json
import os
import random
from pathlib import Path
from datetime import datetime

ANSWERED_FOLDER = Path(os.path.expanduser("~")) / "Downloads" / "llm_resending" / "answered"
OUTPUT_FOLDER   = Path(os.path.expanduser("~")) / "Downloads" / "llm_resending"
SAMPLE_SIZE     = 30

# Load all answered entries across every JSON file
all_entries = []
for fname in sorted(os.listdir(ANSWERED_FOLDER)):
    if not fname.endswith(".json"):
        continue
    with open(ANSWERED_FOLDER / fname, "r", encoding="utf-8") as f:
        entries = json.load(f)
    for entry in entries:
        entry["_source_file"] = fname
    all_entries.extend(entries)

# Only sample from answered entries (skip skipped / timeout / error)
answered_entries = [e for e in all_entries if e.get("status") == "answered"]

print(f"Total entries loaded : {len(all_entries)}")
print(f"Answered entries     : {len(answered_entries)}")

if len(answered_entries) < SAMPLE_SIZE:
    print(f"Warning: only {len(answered_entries)} answered entries available — sampling all of them.")
    sample = answered_entries
else:
    sample = random.sample(answered_entries, SAMPLE_SIZE)

timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
out_path  = OUTPUT_FOLDER / f"sample_{SAMPLE_SIZE}_{timestamp}.json"

with open(out_path, "w", encoding="utf-8") as f:
    json.dump(sample, f, ensure_ascii=False, indent=2)

print(f"Sampled {len(sample)} questions → {out_path}")
