# --- Field validation prompts ---
EVAL_SYSTEM_PROMPT = '''
You are evaluating a data retrieval system for an HR chatbot.
Your job is to assess whether the fields retrieved from an employee database are
necessary and sufficient to answer the given HR policy question.

Available field definitions (for context):
{sap_data_dictionary}

Respond ONLY with a valid JSON object — no markdown, no prose.
'''

EVAL_USER_PROMPT = '''
Question: "{question}"
Fields retrieved: {fields}

Policy chunks used to answer this question:
{policy_context}

Using the policy chunks above as the source of truth, for each retrieved field answer:
1. Is this field needed to answer the question based on what the policy actually says? (true / false)
2. If false, why is it unnecessary?

Then give an overall verdict:
- MINIMAL       : All fields are necessary and none are missing
- OVER-RETRIEVED: Some fields are unnecessary
- UNDER-RETRIEVED: Important fields are missing
- EMPTY         : No fields were retrieved but personalization was needed

If the verdict is UNDER-RETRIEVED or EMPTY, list the additional fields from the Available
field definitions that are required based on conditions or distinctions in the policy chunks,
and explain why each is needed.

Return as JSON with this exact structure:
{{
  "evaluation": {{
    "<field_name>": {{"necessary": true/false, "reason": "<short reason>"}}
  }},
  "verdict": "MINIMAL | OVER-RETRIEVED | UNDER-RETRIEVED | EMPTY",
  "verdict_reason": "<one sentence explaining the verdict>",
  "missing_fields": [
    {{"field": "<field_name>", "reason": "<why this field is needed based on the policy chunks>"}}
  ]
}}

Note: "missing_fields" should be an empty list [] when the verdict is MINIMAL or OVER-RETRIEVED.
'''

def evaluate_fields(az_openai_client, model_name: str, query: str, required_sap_fields: list, retrieval_list: list) -> dict:
    policy_context = "\n\n".join(
        f"[{i+1}] {chunk['title']}\n{chunk['content']}"
        for i, chunk in enumerate(retrieval_list)
    ) if retrieval_list else "No policy chunks were retrieved for this question."

    messages = [
        {
            "role": "system",
            "content": EVAL_SYSTEM_PROMPT.format(sap_data_dictionary=llm_core.SAP_DATA_DICTIONARY),
        },
        {
            "role": "user",
            "content": EVAL_USER_PROMPT.format(
                question=query,
                fields=json.dumps(required_sap_fields),
                policy_context=policy_context,
            ),
        },
    ]
    response = az_openai_client.chat.completions.create(
        model=model_name,
        messages=messages,
        temperature=0,
        max_tokens=1024,
        top_p=1,
        frequency_penalty=0,
        presence_penalty=0,
    )
    raw = response.choices[0].message.content.strip()
    try:
        parsed = json.loads(raw)
    except json.JSONDecodeError:
        parsed = {"raw_response": raw, "parse_error": True}
    return {
        "evaluation":     parsed.get("evaluation", {}),
        "verdict":        parsed.get("verdict", "PARSE_ERROR"),
        "verdict_reason": parsed.get("verdict_reason", ""),
        "missing_fields": parsed.get("missing_fields", []),
    }

def decrypt_value(value: str) -> str:
    if not value:
        return ""
    return fernet.decrypt(value.encode()).decode()


def load_policy_chunks(policy_source: str) -> list[dict]:
    """
    Match a policy_source like 'Appointment.pdf.md' to chunks in output_json/
    where metadata.source == 'Appointment.pdf' (the .md suffix is stripped).
    Returns chunks normalized to the {title, content} shape that llm_core expects.
    """
    if not policy_source:
        return []

    target_source = policy_source[:-3] if policy_source.lower().endswith(".md") else policy_source

    matched: list[dict] = []
    if not OUTPUT_JSON_FOLDER.exists():
        return matched

    for fname in sorted(os.listdir(OUTPUT_JSON_FOLDER)):
        if not fname.endswith(".json"):
            continue
        with open(OUTPUT_JSON_FOLDER / fname, "r", encoding="utf-8") as f:
            chunks = json.load(f)
        for chunk in chunks:
            if chunk.get("metadata", {}).get("source", "") != target_source:
                continue
            section = (chunk.get("section") or "").strip()
            title = section if section else target_source
            matched.append({
                "title":   title,
                "content": chunk.get("content", ""),
            })
    return matched


def load_queries_by_file() -> list[tuple[str, list[dict]]]:
    """
    Returns a list of (filename_stem, queries) — one entry per input file.

    Supports the flat 'questions' format produced for whole-policy question sets:
      {"policy_source": "Appointment.pdf.md", "questions": [{"question_number": 1, "question": "..."}]}
    """
    files = []
    for fname in sorted(os.listdir(QUERIES_FOLDER)):
        if not fname.endswith(".json"):
            continue
        with open(QUERIES_FOLDER / fname, "r", encoding="utf-8") as f:
            data = json.load(f)

        queries = []
        if isinstance(data, dict) and "questions" in data:
            policy_source = data.get("policy_source", "")
            for qa in data.get("questions", []):
                if "question" in qa:
                    queries.append({
                        "query":           qa["question"],
                        "question_number": qa.get("question_number"),
                        "policy_source":   policy_source,
                    })
        elif isinstance(data, list):
            queries.extend([{"query": q} for q in data])
        elif isinstance(data, dict) and "query" in data:
            queries.append({"query": data["query"]})

        if queries:
            files.append((Path(fname).stem, queries))
    return files


def fetch_user_data(required_sap_fields: list[str]) -> dict:
    payload = json.dumps({
        "query": USERNAME,
        "searchType": "vector",
        "vector": {
            "kind": "vector",
            "fields": "user_name_vector",
            "k": 3,
            "exhaustive": True,
        },
        "$select": "user_name, " + ", ".join(required_sap_fields),
        "top": 1
    })
    response = requests.post(SAP_URL, headers=SAP_HEADERS, data=payload)
    if response.status_code != 200:
        print(f"SAP search error {response.status_code}: {response.text}")
        return {}

    result = response.json().get("value", [])
    if not result:
        return {}

    raw = result[0]
    fields_to_decrypt = ["user_name"] + required_sap_fields
    return {f: decrypt_value(raw.get(f, "")) for f in fields_to_decrypt}


def run_pipeline(query: str, azure_config: dict, az_model: str, policy_source: str = "") -> dict:
    # Create a fresh client per question so abandoned connections from timeouts
    # don't exhaust the shared connection pool
    az_openai_client = AzureOpenAI(
        api_key=azure_config['api_key'],
        api_version=azure_config['api_version'],
        azure_endpoint=azure_config['azure_endpoint'],
        timeout=TIMEOUT_SECONDS,
    )

    # Single trace created here in the worker thread so all child spans
    # nest under it correctly
    lf = get_client()
    with lf.start_as_current_observation(as_type="span", name="QnA") as start_trace:

        # Step 1 — Pull every chunk belonging to this policy (matched by source)
        retrieval_list = load_policy_chunks(policy_source)
        if not retrieval_list:
            raise ValueError(f"No chunks found in output_json for policy_source={policy_source!r}")

        # Step 2 — Identify required SAP fields from the policy chunks
        initial_required_fields = llm_core.identify_required_personalization_fields(
            az_openai_client=az_openai_client,
            model_name=az_model,
            retrieval_list=retrieval_list,
            start_trace=start_trace,
        )
        if isinstance(initial_required_fields, str):
            initial_required_fields = ast.literal_eval(initial_required_fields)
        initial_required_fields = list(initial_required_fields)

        # Step 2b — Validate retrieved SAP fields against the policy chunks
        field_validation = evaluate_fields(
            az_openai_client=az_openai_client,
            model_name=az_model,
            query=query,
            required_sap_fields=initial_required_fields,
            retrieval_list=retrieval_list,
        )
        verdict = field_validation.get("verdict", "")
        print(f"  Field validation verdict: {verdict}")

        # Step 2c — Reconcile field list based on validation
        final_required_fields = list(initial_required_fields)
        if verdict in ("OVER-RETRIEVED", "UNDER-RETRIEVED", "EMPTY"):
            unnecessary = {
                field for field, detail in field_validation.get("evaluation", {}).items()
                if not detail.get("necessary", True)
            }
            final_required_fields = [f for f in final_required_fields if f not in unnecessary]

            for mf in field_validation.get("missing_fields", []):
                field = mf.get("field", "")
                if field and field not in final_required_fields:
                    final_required_fields.append(field)

            print(f"  Reconciled fields: {final_required_fields}")

        # Step 3 — Fetch + decrypt employee data using the reconciled field list
        employee_data = {}
        if final_required_fields:
            employee_data = fetch_user_data(final_required_fields)

        # Step 4 — Personalize response using the reconciled fields
        personalized_text = llm_core.personalize_response(
            az_openai_client=az_openai_client,
            model_name=az_model,
            query=query,
            retrieval_list=retrieval_list,
            employee_data=employee_data,
            start_trace=start_trace,
        )

        start_trace.update_trace(
            input=query,
            output=personalized_text,
            user_id='pipeline',
        )

    return {
        "initial_required_fields": initial_required_fields,
        "verdict":                 verdict,
        "verdict_reason":          field_validation.get("verdict_reason", ""),
        "evaluation":              field_validation.get("evaluation", {}),
        "missing_fields":          field_validation.get("missing_fields", []),
        "final_required_fields":   final_required_fields,
        "answer":                  personalized_text,
    }


def main():
    config = Config()
    os.environ["LANGFUSE_PUBLIC_KEY"] = config.get('tracing', 'LANGFUSE_PUBLIC_KEY')
    os.environ["LANGFUSE_SECRET_KEY"] = config.get('tracing', 'LANGFUSE_SECRET_KEY')
    os.environ["LANGFUSE_HOST"]       = config.get('tracing', 'LANGFUSE_HOST')

    lf = get_client()

    azure_config = {
        'api_key':        config.config['AZURE_OPENAI_API_KEY'],
        'api_version':    config.get('models', 'AZURE_OPENAI_API_VERSION'),
        'azure_endpoint': config.get('models', 'AZURE_OPENAI_ENDPOINT'),
    }
    az_model = config.get('models', 'AZURE_OPENAI_CHAT_DEPLOYMENT')

    files = load_queries_by_file()
    total_files = len(files)
    print(f"Loaded {total_files} input file(s)\n")

    timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")

    for file_idx, (fname_stem, queries) in enumerate(files, 1):
        print(f"=== File {file_idx}/{total_files}: {fname_stem}.json ({len(queries)} questions) ===\n")

        answered = []
        out_path = ANSWERED_FOLDER / f"{fname_stem}_{timestamp}.json"

        for item in queries:
            query = item["query"]
            policy_source = item.get("policy_source", "")
            print(f"[{policy_source} | Q{item.get('question_number','')}]")
            print(f"Q: {query}")

            if not policy_source:
                print(f"  → Skipped: question has no policy_source.\n{'-'*60}\n")
                answered.append({
                    "policy_source":           policy_source,
                    "question_number":         item.get("question_number"),
                    "question":                query,
                    "initial_required_fields": [],
                    "verdict":                 "",
                    "verdict_reason":          "",
                    "evaluation":              {},
                    "missing_fields":          [],
                    "final_required_fields":   [],
                    "answer":                  "",
                    "status":                  "skipped",
                    "error":                   "No policy_source provided — cannot match policy chunks",
                })
                with open(out_path, "w", encoding="utf-8") as f:
                    json.dump(answered, f, ensure_ascii=False, indent=2)
                print(f"Saved {len(answered)}/{len(queries)} → {out_path}")
                continue

            error_message = ""
            result: dict = {}
            try:
                executor = ThreadPoolExecutor(max_workers=1)
                future = executor.submit(
                    run_pipeline, query, azure_config, az_model, policy_source
                )
                try:
                    result = future.result(timeout=TIMEOUT_SECONDS)
                finally:
                    executor.shutdown(wait=False)

                lf.flush()

                print(f"A: {result.get('answer','')}\n{'-'*60}\n")
                status = "answered"

            except FuturesTimeoutError:
                print(f"TIMEOUT after {TIMEOUT_SECONDS}s — skipping question.\n{'-'*60}\n")
                status = "timeout"
                error_message = f"TimeoutError: exceeded {TIMEOUT_SECONDS}s"

            except Exception as e:
                print(f"ERROR: {e} — skipping question.\n{'-'*60}\n")
                status = "error"
                error_message = f"{type(e).__name__}: {e}"

            answered.append({
                "policy_source":           policy_source,
                "question_number":         item.get("question_number"),
                "question":                query,
                "initial_required_fields": result.get("initial_required_fields", []),
                "verdict":                 result.get("verdict", ""),
                "verdict_reason":          result.get("verdict_reason", ""),
                "evaluation":              result.get("evaluation", {}),
                "missing_fields":          result.get("missing_fields", []),
                "final_required_fields":   result.get("final_required_fields", []),
                "answer":                  result.get("answer", ""),
                "status":                  status,
                "error":                   error_message,
            })

            with open(out_path, "w", encoding="utf-8") as f:
                json.dump(answered, f, ensure_ascii=False, indent=2)
            print(f"Saved {len(answered)}/{len(queries)} → {out_path}")

        print(f"\nFile done. {sum(1 for a in answered if a['status'] == 'answered')} answered, "
              f"{sum(1 for a in answered if a['status'] == 'timeout')} timed out, "
              f"{sum(1 for a in answered if a['status'] == 'error')} errored.\n")


if __name__ == "__main__":
    main()
