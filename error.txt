measuring_point = "HalfHourlyConsumption_HTA_2025-06-01_2025-07-01"
data_dict_temp = {measuring_point: data_dict[measuring_point]}

tz = data_dict_temp[measuring_point].index[0].tz
print("tz:", tz)

ad_detector = AutoAD(
    namespace="dsta-be28",
    data_frequency=preprocessor.data_frequency,
    method="statistic",
    unit_of_measurement="Wh",
    lookback_period=7
)

detection_date = pd.Timestamp("2025-06-17")
end_date = pd.Timestamp("2025-06-17")

if tz is not None:
    detection_date = detection_date.tz_localize(tz)
    end_date = end_date.tz_localize(tz)

start_date = data_dict_temp[measuring_point].index.min()
print("data starts at:", start_date)

required_start = detection_date - pd.Timedelta(days=ad_detector.lookback_period)
print("required history starts at:", required_start)

if required_start < start_date:
    raise ValueError(
        f"Not enough data for lookback period={ad_detector.lookback_period}. "
        f"Need data from {required_start}, but only have data from {start_date}."
    )

while detection_date <= end_date:
    score_df_dict = ad_detector.detect(
        data_dict=data_dict_temp,
        detect_date=detection_date
    )
    ad_detector.top_k(
        k=1,
        sort_by="deviation_percentage",
        report=True,
        plot=True,
        dev_perc_threshold=5
    )
    detection_date += pd.Timedelta(1, unit="d")
