SAP_SYSTEM_PROMPT = '''
<Role>
You are a system that determines exactly which employee data fields are required
to give a personalised answer to an HR policy question.
</Role>

<Instructions>
You will receive an employee's question and the relevant HR policy documents.

Step 1 — Understand the question: What specific aspect of the policy is the employee asking about?

Step 2 — Scan the policy for employee-specific conditions: Find every place the policy makes a
distinction based on an employee attribute (e.g. "executives receive X", "part-time employees are
entitled to Y", "employees below grade Z are eligible").

Step 3 — Filter to what is relevant to the question: Drop any condition that belongs to a part of
the policy the employee did not ask about.

Step 4 — Map to available fields: For each remaining condition, find the matching field from the
Available User Data Fields list. Match on meaning, not wording.
  Example: "executive / non-executive" → `employee_sub_group`, NOT `generic_role`

Step 5 — Self-check each field: Keep a field only if ALL three are true:
  (a) The policy text explicitly uses this attribute as a condition or variable
  (b) The condition is directly relevant to the employee's question
  (c) The answer would differ depending on the employee's value for this field
  Drop any field that fails even one check.

Step 6 — Output: Return ONLY a Python list of field name strings. No explanation, no prose, no markdown.
  - If no personalisation is needed, return []
  - Valid example: ["employee_sub_group", "full_time_part_time"]
</Instructions>

Available User Data Fields:
{sap_data_dictionary}
'''

def run_conversation_core(
        *,
        messages,
        system_content,
        openai_client,
        model_name,
        question_section: str = "",
        policy_source: str = "",
    ):
    """
    Generator that yields:
      - ("chunk", text)
      - ("done", completion_text, trace_metric_list, doc_dict)
    """

    _client = openai_client
    _model = model_name

    doc_dict = {}
    retrieval_list = []

    # Prepare messages for the LLM
    print("Before entering dcc_prepare_chat_messages", flush=True)
    messages["messages"] = dcc_prepare_chat_messages(
        messages["messages"],
        system_content,
        doc_dict=doc_dict,
        retrieval_list_out=retrieval_list,
        question_section=question_section,
        policy_source=policy_source,
    )

    with lf.start_as_current_observation(
        as_type="generation",
        name="Answer",
        model=_model
    ) as answer_span:

        completion_text = ""

        response = _client.chat.completions.create(
            messages=messages["messages"],
            model=_model,
            temperature=0,
            max_tokens=4096,
            top_p=1,
            frequency_penalty=0,
            presence_penalty=0,
            stream=True,
        )

        for event in response:
            if not event.choices:
                continue

            if event.choices[0].finish_reason is not None:
                break

            if event.choices[0].delta.content:
                chunk = event.choices[0].delta.content
                completion_text += chunk
                yield ("chunk", chunk)

        trace_metric_list = create_trace_metric_list(
            "Answer",
            messages["messages"],
            completion_text
        )

        update_span_list(answer_span,trace_metric_list)
        
        yield ("done", completion_text, trace_metric_list, doc_dict, retrieval_list)

def build_reference_appendix(completion_text: str, doc_dict: dict) -> str:
    if not doc_dict:
        return ""

    context_numbers = re.findall(r'\[Ref (\d+)', completion_text)
    sources_used = [
        v['text']
        for v in doc_dict.values()
        if v['index'] in map(int, context_numbers)
    ]

    if not sources_used:
        return ""

    return (
        "\n\n__References__ Used:\n"
        + "\n".join(sources_used)
        + f"\n\n{DISCLAIMER}"
    )

def identify_required_personalization_fields(
    az_openai_client,
    model_name,
    retrieval_list: list,
    start_trace,
    query: str = "",
) -> list[str]:
    print("entered identify")
    identify_SAP_fields_span = start_trace.start_generation(name="Identify_SAP_fields")
    policy_context = "\n\n".join(
        f"[{i+1}] {chunk['title']}\n{chunk['content']}"
        for i, chunk in enumerate(retrieval_list)
    ) if retrieval_list else "No policy chunks retrieved."

    messages = [
        {"role": "system", "content": SAP_SYSTEM_PROMPT.format(sap_data_dictionary=SAP_DATA_DICTIONARY)},
        {"role": "user", "content": SAP_USER_PROMPT.format(query=query, policy_context=policy_context)},
    ]

    response = az_openai_client.chat.completions.create(
        model=model_name,
        messages=messages,
        temperature=0,
        max_tokens=512,
        top_p=1,
        frequency_penalty=0,
        presence_penalty=0,
    )
    raw = response.choices[0].message.content.strip()

    trace_metric_list = create_trace_metric_list("Identify_SAP_fields", messages, raw)
    update_span_list(identify_SAP_fields_span, trace_metric_list)
    identify_SAP_fields_span.end()

    print(f"Identify required fields: {raw}")

    try:
        fields = ast.literal_eval(raw)
        if isinstance(fields, list):
            return fields
    except (ValueError, SyntaxError):
        pass

    match = re.search(r'\[.*?\]', raw, re.DOTALL)
    if match:
        try:
            fields = ast.literal_eval(match.group())
            if isinstance(fields, list):
                return fields
        except (ValueError, SyntaxError):
            pass

    print("  WARNING: could not parse field list — defaulting to []")
    return []

def personalize_response(
    az_openai_client,
    model_name: str,
    query: str,
    retrieval_list: list,
    employee_data: dict,
    start_trace,
) -> str:
    personalize_span = start_trace.start_generation(name="Personalize_Response")
    policy_context = "\n\n".join(
        f"[{i+1}] {chunk['title']}\n{chunk['content']}"
        for i, chunk in enumerate(retrieval_list)
    )
    filtered_data = {k: v for k, v in employee_data.items() if k not in PII_FIELDS and v}
    employee_data_str = "\n".join(f"- {k}: {v}" for k, v in filtered_data.items())
    messages = [
        {"role": "system", "content": PERSONALIZE_SYSTEM_PROMPT},
        {"role": "user", "content": PERSONALIZE_USER_PROMPT.format(
            query=query,
            policy_context=policy_context,
            employee_data=employee_data_str,
        )},
    ]
    openai_response = az_openai_client.chat.completions.create(
        model=model_name,
        messages=messages,
        max_tokens=4096,
        temperature=0,
        top_p=1,
        frequency_penalty=0,
        presence_penalty=0,
        stream=True,
    )
    personalized_text = ""
    for event in openai_response:
        if not event.choices:
            continue
        if event.choices[0].finish_reason is not None:
            break
        if event.choices[0].delta.content:
            personalized_text += event.choices[0].delta.content

    trace_metric_list = create_trace_metric_list("Personalize_Response", query, personalized_text)
    update_span_list(personalize_span, trace_metric_list)
    personalize_span.end()
    return personalized_text
