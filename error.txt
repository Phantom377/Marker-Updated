import json
import os
import random
from pathlib import Path
from datetime import datetime

VALIDATION_FOLDER = Path(os.path.expanduser("~")) / "Downloads" / "llm_resending" / "validation"
OUTPUT_FOLDER     = Path(os.path.expanduser("~")) / "Downloads" / "llm_resending"
SAMPLE_SIZE       = 30

# Load all entries across every validation JSON file
all_entries = []
for fname in sorted(os.listdir(VALIDATION_FOLDER)):
    if not fname.endswith(".json"):
        continue
    with open(VALIDATION_FOLDER / fname, "r", encoding="utf-8") as f:
        entries = json.load(f)
    for entry in entries:
        entry["_source_file"] = fname
    all_entries.extend(entries)

# Skip entries where the LLM response failed to parse
valid_entries = [e for e in all_entries if not e.get("_parse_error", False)]

print(f"Total entries loaded : {len(all_entries)}")
print(f"Valid entries        : {len(valid_entries)} (excluding parse errors)")

if len(valid_entries) < SAMPLE_SIZE:
    print(f"Warning: only {len(valid_entries)} valid entries available — sampling all of them.")
    sample = valid_entries
else:
    sample = random.sample(valid_entries, SAMPLE_SIZE)

timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
out_path  = OUTPUT_FOLDER / f"sample_{SAMPLE_SIZE}_{timestamp}.json"

with open(out_path, "w", encoding="utf-8") as f:
    json.dump(sample, f, ensure_ascii=False, indent=2)

print(f"Sampled {len(sample)} questions → {out_path}")
