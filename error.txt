def DB_search(query)->list[dict]:
    # start Langfuse tracing of DBSearch
    with lf.start_as_current_observation(
        as_type="retriever",
        name="DBSearch"
    ) as DBSearch_span:
        retrieval_list = []

        payload = json.dumps({
            "query": query,
            "searchType": "hybrid",
            "vector": {
                "kind": "vector",
                "fields": "page_contentVector",
                "k": 10,
                "exhaustive": True,
                "queryType": "semantic",
                "semanticConfiguration": "hrbuddy-semantic-config",
                "select": "section, source, link, page_content, tags, created_at",
                "searchFields" : "page_content", # optional, without searchFields will search all searchable fields, with searchFields it will only search in the specified fields
                "filter": "project eq 'HRBUDDY' and tags/any(t: t eq 'current')",
                "top": 5
            }
        })

        headers = {
            'Content-Type': 'application/json',
            'model': 'openai',
            'subModel': config.get('vector_database', 'embedding_model'),
            'client-id': config.get('vector_database', 'client_id'),
            'client-secret': config.get('vector_database', 'client_secret'),
            'project': 'HRBuddy',
            'vector-index': 'spl_hr_buddy'
        }
        
        try:
            response = requests.request("POST", config.get('vector_database', 'url'), headers=headers, data=payload)
            results = json.loads(response.text).get('value')

            for result in results:
                retrieval_dict = {}
                retrieval_dict['title'] = result.get('source')
                retrieval_dict['link'] = result.get('link')
                # retrieval_dict['score'] = result.get('@search.score')
                retrieval_dict['content'] = result.get('page_content')
                retrieval_dict['date_updated'] = result.get('date_updated', str(datetime(9999, 1, 1)))

                retrieval_list.append(retrieval_dict)
        except Exception as e:
            print(f"Error found: {e}")
            return []

        # logger.warning(f'azure ai search query time: {time.time()-marker:.2f}')
        # trace_metric_list = create_trace_metric_list('QueryRewrite', query, retrieval_list)
        # update_span_list(DBSearch_span,trace_metric_list)
        DBSearch_span.update(input=query, output=retrieval_list)
        # DBSearch_span.end()
        return retrieval_list # returns a list of Dictionaries. Each Dictionary is a result of the search

def query_rewrite(msg_list):

    # start Langfuse tracing of user query rewrite
    
    with lf.start_as_current_observation(
        as_type="generation",
        name="QueryRewrite",
        model=config.get('OpenAI', 'QUERY_REWRITE_MODEL')
    ) as rewrite_span:

        if len(msg_list) <= 1:
            logger.warning(f"Skipping query rewrite: {msg_list[-1]['content']}")
            # rewrite_span.end()
            return
        
        _query_list = []
        for m in msg_list:
            if m['role'] == 'system':
                continue
            elif m['role'] == 'assistant':
                _query_list.append(f"{m['role']}: {m['content'].split('__References__')[0]}")
            else:
                _query_list.append(f"{m['role']}: {m['content']}")

        # Query Rewrite
        rewrite_msg_list = [
            {
                'role': 'system',
                'content': get_prompt_template('SYS_PROMPT_REWRITE')
            },
            {
                'role': 'user',
                'content': '\n'.join(_query_list)
            }
        ]
        response = client.chat.completions.create(
            messages=rewrite_msg_list,
            model=config.get('OpenAI', 'MODEL'),
            temperature=0,
            max_tokens=500
        )
        new_query_str = response.choices[0].message.content

        msg_list[-1]['content'] = new_query_str
        # trace_metric_list = create_trace_metric_list('QueryRewrite',rewrite_msg_list,new_query_str)
        # update_span_list(rewrite_span,trace_metric_list)
        rewrite_span.update(input=rewrite_msg_list, output=new_query_str)

        return 


def dcc_prepare_chat_messages(
        msg_list,
        system_content,
        doc_dict = {}
    ):
    '''
    For chat, rewrite the query string, then retrieve contexts from DBSearch 
    and return the updated messages.
    '''
    # Rewrite query string
    query_rewrite(msg_list)
    query_str = msg_list[-1]['content']

    # Retrieve Contexts
    retrieval_list = DB_search(query_str)
  
    ind = 0
    for i in retrieval_list:
        if i["title"] not in doc_dict:
            ind += 1
            # alink = '<a href="'+i["link"]+'" target="_blank">'+i["title"]+'</a>'
            doc_dict[i['title']] = {
                'text': f'* [{ind}] [{i["title"]}]({i["link"]}) - Last Updated: {datetime.fromisoformat(i["date_updated"]).strftime("%d %b %Y")}',
                #'text': f'* [{ind}] {alink}',
                'index': ind
            }

    # Prepare new context
    context_str = ''
    for i in retrieval_list:
        # if i['score'] < 0.01:
        #     continue
        context_str += get_prompt_template('CONTEXT_TEMPLATE',
                                            counter=doc_dict[i['title']]['index'],
                                            title=i['title'],
                                            content=i['content']
                                            )

    _query_list = []
    for m in msg_list:
        if m['role'] == 'system':
            continue
        elif m['role'] == 'assistant':
            _query_list.append(
                f"{m['role']}: {m['content'].split('__References__')[0]}"
            )
        else:
            _query_list.append(
                f"{m['role']}: {m['content']}"
            )
    new_query_str = get_prompt_template('RAG_PROMPT_TEMPLATE',context_str=context_str,query_str='\n'.join(_query_list))

    # Update query
    return  (
        [
            {"role": "system", "content": system_content},
            {"role": "user", "content": new_query_str},
        ]
    )


class PredictiveQuestion(BaseModel):
    probability: float = Field(description="probability")
    question: str = Field(description="question")


class PredictiveQuestions(BaseModel):
    questions: List[PredictiveQuestion] = Field(description="list of predictive questions")


def get_predictive_question_generator(llm):

    # output_parser = PydanticToolsParser(tools=[PredictiveQuestions])

    prompt = ChatPromptTemplate.from_messages(
        [
            ("system", get_prompt_template('SYS_PROMPT_PREDICTIVE')),
            ("user", "{context}"),
        ]
    )

    structured_llm = llm.with_structured_output(PredictiveQuestions)

    predictor = {"context": RunnablePassthrough()} | prompt | structured_llm
    return predictor

# Creates score on Langfuse for tracing user feedback
def trace_score(dcc_config, trace_id, score, comment):
    os.environ["LANGFUSE_PUBLIC_KEY"] = dcc_config.get('tracing','LANGFUSE_PUBLIC_KEY')
    os.environ["LANGFUSE_SECRET_KEY"] = dcc_config.get('tracing','LANGFUSE_SECRET_KEY')
    os.environ["LANGFUSE_HOST"] = dcc_config.get('tracing','LANGFUSE_HOST')

    lf = get_client()

    lf.create_score(
        trace_id=trace_id,
        name="feedback",
        value=score,
        comment=comment,
    )

    lf.flush()

# Creates a default metric list for tracing
def create_trace_metric_list(name,input,output):
    trace_metric_list = {
        'name': name,
        'input': input,
        'output': output,
        'model': 'gpt-4.1',
        'metadata': {
            "provider": "azure",
            "model_name": "gpt-4.1",
            "model_type": "chat",
        },
    }
    return trace_metric_list

# LLM call for title generation
def GenerateTitle(chat_client,prompt,message,conversation_id,request,messages,title_span ):
    response, token_usage = chat_client.chat([
                SystemMessage(content=get_prompt_template('SYS_GEN_TITLE_PROMPT')),
                HumanMessage(content=prompt+message.message)
            ])
    title = response.strip().replace('"', '')
    
    trace_metric_list = create_trace_metric_list('GenerateTitle',messages,title)
    update_span_list(title_span,trace_metric_list)

    title_span.update_trace(
        name='Title',
        session_id=f"cnv-{conversation_id}",
        user_id=request.user.username,
        input=message.message,
        output=title
    )
    return token_usage, title

# Updates span with default args
def update_span_list(span,trace_metric_list):
    span.update(
        input= trace_metric_list['input'],
        output= trace_metric_list['output'],
        model = trace_metric_list['model'],
        metadata = trace_metric_list['metadata'],
    )
