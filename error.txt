import json
import os
import sys
from pathlib import Path

# --- Adjust these paths to match your environment ---
PROJECT_ROOT = Path(os.path.expanduser("~")) / "hr-chatbot" / "src"
sys.path.insert(0, str(PROJECT_ROOT))
sys.path.insert(0, str(PROJECT_ROOT / "server"))

from server.model_source.config import Config
from openai import AzureOpenAI

ANSWERED_FOLDER = Path(os.path.expanduser("~")) / "Downloads" / "llm_resending" / "answered"
VALIDATION_FOLDER = Path(os.path.expanduser("~")) / "Downloads" / "llm_resending" / "validation"
VALIDATION_FOLDER.mkdir(exist_ok=True)

config = Config()
az_client = AzureOpenAI(
    api_key=config.config['AZURE_OPENAI_API_KEY'],
    api_version=config.get('models', 'AZURE_OPENAI_API_VERSION'),
    azure_endpoint=config.get('models', 'AZURE_OPENAI_ENDPOINT'),
    timeout=60,
)
AZ_MODEL = config.get('models', 'AZURE_OPENAI_CHAT_DEPLOYMENT')

# Load all answered JSON files — skip entries with no fields retrieved (empty list)
all_entries = []
for fname in sorted(os.listdir(ANSWERED_FOLDER)):
    if not fname.endswith(".json"):
        continue
    with open(ANSWERED_FOLDER / fname, "r", encoding="utf-8") as f:
        entries = json.load(f)
    for entry in entries:
        entry["_file"] = fname
    all_entries.extend(entries)

# Only validate answered entries that actually retrieved at least one field
to_validate = [
    e for e in all_entries
    if e.get("status") == "answered" and e.get("required_fields")
]

print(f"Total entries loaded  : {len(all_entries)}")
print(f"Entries to validate   : {len(to_validate)} (answered with ≥1 field retrieved)")
