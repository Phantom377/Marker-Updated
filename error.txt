# --- Test 1: Query alone (no retrieval, no chunks) ---
print("=" * 60)
print("[Test 1] Query sent directly to LLM with no retrieved chunks...")
try:
    response = az_openai_client.chat.completions.create(
        model="gpt-4o",
        messages=[
            {"role": "system", "content": system_content},
            {"role": "user",   "content": debug_query},
        ],
        max_tokens=100,
        temperature=0,
    )
    print(f"  ✓ OK — response: {response.choices[0].message.content[:100]}")
