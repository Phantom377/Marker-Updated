import server.chat.llm_core as llm_core
import server.model_source.dcc_ai_search_json as dcc_ai_search_json
from server.model_source.config import Config
from openai import OpenAI, AzureOpenAI
from langfuse import get_client

import importlib
importlib.reload(llm_core)

config = Config()
lf = get_client()
system_content = get_prompt_template('SYS_PROMPT')
debug_query = "I am going to have my first child soon. What leave can I get?"

# Use AzureOpenAI directly (bypasses gateway PII guardrail)
az_client = AzureOpenAI(
    api_key=config.config['AZURE_OPENAI_API_KEY'],
    api_version=config.get('models', 'AZURE_OPENAI_API_VERSION'),
    azure_endpoint=config.get('models', 'AZURE_OPENAI_ENDPOINT'),
)
az_model = config.get('models', 'AZURE_OPENAI_CHAT_DEPLOYMENT')

# --- Test 1: Query + system prompt only (no chunks) through Azure ---
print("=" * 60)
print("[Test 1] Query + system prompt through AZURE (no chunks)...")
try:
    response = az_client.chat.completions.create(
        model=az_model,
        messages=[
            {"role": "system", "content": system_content},
            {"role": "user",   "content": debug_query},
        ],
        max_tokens=100,
        temperature=0,
    )
    print(f"  ✓ OK — response: {response.choices[0].message.content[:100]}")
except Exception as e:
    print(f"  ✗ FAILED\n  Error: {e}")
