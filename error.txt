def _load_all_chunks() -> list[dict]:
    """Load all chunks from every JSON file in output_json."""
    all_chunks = []
    for fname in os.listdir(JSON_FOLDER):
        if fname.endswith(".json"):
            with open(os.path.join(JSON_FOLDER, fname), "r", encoding="utf-8") as f:
                chunks = json.load(f)
            all_chunks.extend(chunks)
    return all_chunks

def _normalize_section(section: str) -> str:
    """Strip markdown bold markers, collapse whitespace, and lowercase for comparison."""
    text = re.sub(r'\*+', '', section)   # remove ** markers
    text = re.sub(r'\s+', ' ', text)     # collapse double spaces left behind
    return text.strip().lower()

def DB_search(query: str, question_section: str = "") -> list[dict]:
    with lf.start_as_current_observation(
        as_type="retriever",
        name="DBSearch"
    ) as DBSearch_span:
        retrieval_list = []
        try:
            all_chunks = _load_all_chunks()
            q_sec = _normalize_section(question_section)

            # If the question has no section, skip retrieval — handled upstream in pipeline
            if not q_sec:
                print("DB_search: question has no section — skipping retrieval")
                DBSearch_span.update(input=query, output=[])
                return []

            # Exact match after stripping markdown bold markers and whitespace
            # Also skips chunks whose section is empty after normalization
            matched = [
                c for c in all_chunks
                if _normalize_section(c.get("section", "")) == q_sec
            ]

            if not matched:
                print(f"DB_search: no chunk exactly matched section '{question_section}'")
                DBSearch_span.update(input=query, output=[])
                return []

            # If multiple chunks share the same section, take the first one
            best_chunk = matched[0]
            print(f"DB_search: matched {len(matched)} chunk(s) for section '{question_section}' — using chunk_index {best_chunk.get('chunk_index', '?')}")

            source = best_chunk.get("metadata", {}).get("source", "Unknown")
            retrieval_list.append({
                "title":        source,
                "section":      best_chunk.get("section", ""),
                "link":         best_chunk.get("metadata", {}).get("link", ""),
                "content":      best_chunk.get("content", ""),
                "context":      best_chunk.get("context", ""),
                "date_updated": str(datetime(9999, 1, 1)),
            })
            print(f"DB_search: matched chunk → [{source}] {best_chunk.get('section', '')}")

        except Exception as e:
            print(f"Error found: {e}")
            return []

        DBSearch_span.update(input=query, output=retrieval_list)
        return retrieval_list

def query_rewrite(msg_list):

    # start Langfuse tracing of user query rewrite
    
    with lf.start_as_current_observation(
        as_type="generation",
        name="QueryRewrite",
        model=config.get('OpenAI', 'QUERY_REWRITE_MODEL')
    ) as rewrite_span:

        if len(msg_list) <= 1:
            logger.warning(f"Skipping query rewrite: {msg_list[-1]['content']}")
            # rewrite_span.end()
            return
        
        _query_list = []
        for m in msg_list:
            if m['role'] == 'system':
                continue
            elif m['role'] == 'assistant':
                _query_list.append(f"{m['role']}: {m['content'].split('__References__')[0]}")
            else:
                _query_list.append(f"{m['role']}: {m['content']}")

        # Query Rewrite
        rewrite_msg_list = [
            {
                'role': 'system',
                'content': get_prompt_template('SYS_PROMPT_REWRITE')
            },
            {
                'role': 'user',
                'content': '\n'.join(_query_list)
            }
        ]
        response = client.chat.completions.create(
            messages=rewrite_msg_list,
            model=config.get('OpenAI', 'MODEL'),
            temperature=0,
            max_tokens=500
        )
        new_query_str = response.choices[0].message.content

        msg_list[-1]['content'] = new_query_str
        # trace_metric_list = create_trace_metric_list('QueryRewrite',rewrite_msg_list,new_query_str)
        # update_span_list(rewrite_span,trace_metric_list)
        rewrite_span.update(input=rewrite_msg_list, output=new_query_str)

        return 


def dcc_prepare_chat_messages(
        msg_list,
        system_content,
        doc_dict = {},
        retrieval_list_out = None,
        question_section: str = "",
    ):
    '''
    For chat, retrieve the chunk whose section matches question_section and
    return the updated messages. If no section match is found, the context
    will be empty and the LLM is instructed not to hallucinate.
    '''
    query_str = msg_list[-1]['content']

    # Retrieve context — matched by section header
    retrieval_list = DB_search(query_str, question_section=question_section)
    if retrieval_list_out is not None:
        retrieval_list_out.extend(retrieval_list)
  
    ind = 0
    for i in retrieval_list:
        if i["title"] not in doc_dict:
            ind += 1
            # alink = '<a href="'+i["link"]+'" target="_blank">'+i["title"]+'</a>'
            doc_dict[i['title']] = {
                'text': f'* [{ind}] [{i["title"]}]({i["link"]}) - Last Updated: {datetime.fromisoformat(i["date_updated"]).strftime("%d %b %Y")}',
                #'text': f'* [{ind}] {alink}',
                'index': ind
            }

    # Prepare new context
    context_str = ''
    has_empty_section = False
    for i in retrieval_list:
        context_str += get_prompt_template('CONTEXT_TEMPLATE',
                                            counter=doc_dict[i['title']]['index'],
                                            title=i['title'],
                                            content=i['content']
                                            )
        if not i.get('section', '').strip():
            has_empty_section = True

    # If the retrieved chunk has no section header, warn the LLM not to hallucinate
    if has_empty_section:
        context_str += (
            "\n\n[INSTRUCTION: The retrieved content above does not correspond to a specific "
            "policy section. If it is insufficient to answer the query, respond with: "
            "'I'm sorry, I don't have enough information to answer this question. "
            "Please contact HR directly for assistance.' Do not make assumptions or invent details.]"
        )

    _query_list = []
    for m in msg_list:
        if m['role'] == 'system':
            continue
        elif m['role'] == 'assistant':
            _query_list.append(
                f"{m['role']}: {m['content'].split('__References__')[0]}"
            )
        else:
            _query_list.append(
                f"{m['role']}: {m['content']}"
            )
    new_query_str = get_prompt_template('RAG_PROMPT_TEMPLATE', context_str=context_str, query_str='\n'.join(_query_list))

    # Update query
    return (
        [
            {"role": "system", "content": system_content},
            {"role": "user", "content": new_query_str},
        ]
    )


class PredictiveQuestion(BaseModel):
    probability: float = Field(description="probability")
    question: str = Field(description="question")


class PredictiveQuestions(BaseModel):
    questions: List[PredictiveQuestion] = Field(description="list of predictive questions")


def get_predictive_question_generator(llm):

    # output_parser = PydanticToolsParser(tools=[PredictiveQuestions])

    prompt = ChatPromptTemplate.from_messages(
        [
            ("system", get_prompt_template('SYS_PROMPT_PREDICTIVE')),
            ("user", "{context}"),
        ]
    )

    structured_llm = llm.with_structured_output(PredictiveQuestions)

    predictor = {"context": RunnablePassthrough()} | prompt | structured_llm
    return predictor

# Creates score on Langfuse for tracing user feedback
def trace_score(dcc_config, trace_id, score, comment):
    os.environ["LANGFUSE_PUBLIC_KEY"] = dcc_config.get('tracing','LANGFUSE_PUBLIC_KEY')

# Creates a default metric list for tracing

    return trace_metric_list

# LLM call for title generation
def GenerateTitle(chat_client,prompt,message,conversation_id,request,messages,title_span ):
    response, token_usage = chat_client.chat([
                SystemMessage(content=get_prompt_template('SYS_GEN_TITLE_PROMPT')),
                HumanMessage(content=prompt+message.message)
            ])
    title = response.strip().replace('"', '')
    
    trace_metric_list = create_trace_metric_list('GenerateTitle',messages,title)
    update_span_list(title_span,trace_metric_list)

    title_span.update_trace(
        name='Title',
        session_id=f"cnv-{conversation_id}",
        user_id=request.user.username,
        input=message.message,
        output=title
    )
    return token_usage, title

# Updates span with default args

    span.update(
        input= trace_metric_list['input'],
        output= trace_metric_list['output'],
        model = trace_metric_list['model'],
        metadata = trace_metric_list['metadata'],
    )
