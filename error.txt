def run_conversation_core(
        *,
        messages,
        system_content,
    ):
    """
    Generator that yields:
      - ("chunk", text)
      - ("done", completion_text, trace_metric_list, doc_dict)
    """

    doc_dict = {}

    # Prepare messages for the LLM
    print("Before entering dcc_prepare_chat_messages", flush=True)
    messages["messages"] = dcc_prepare_chat_messages(
        messages["messages"],
        system_content,
        doc_dict=doc_dict
    )

    with lf.start_as_current_observation(
        as_type="generation",
        name="Answer",
        model=config.get('OpenAI', 'MODEL')
    ) as answer_span:
        
        completion_text = ""

        response = client.chat.completions.create(
            messages=messages["messages"],
            model=config.get('OpenAI', 'MODEL'),
            temperature=0,
            max_tokens=4096,
            top_p=1,
            frequency_penalty=0,
            presence_penalty=0,
            stream=True,
        )

        for event in response:
            if not event.choices:
                continue

            if event.choices[0].finish_reason is not None:
                break

            if event.choices[0].delta.content:
                chunk = event.choices[0].delta.content
                completion_text += chunk
                yield ("chunk", chunk)

        trace_metric_list = create_trace_metric_list(
            "Answer",
            messages["messages"],
            completion_text
        )

        update_span_list(answer_span,trace_metric_list)
        
        yield ("done", completion_text, trace_metric_list, doc_dict)

        # yield (required_sap_fields)
        # sap_data = fetch_user_personalization_data(user_id=123, required_fields=required_sap_fields)

def build_reference_appendix(completion_text: str, doc_dict: dict) -> str:
    if not doc_dict:
        return ""

    context_numbers = re.findall(r'\[Ref (\d+)', completion_text)
    sources_used = [
        v['text']
        for v in doc_dict.values()
        if v['index'] in map(int, context_numbers)
    ]

    if not sources_used:
        return ""

    return (
        "\n\n__References__ Used:\n"
        + "\n".join(sources_used)
        + f"\n\n{DISCLAIMER}"
    )
