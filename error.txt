# --- Test 1: Query + system prompt only (no chunks) through gateway ---
print("=" * 60)
print("[Test 1] Query + system prompt through GATEWAY (no chunks)...")
try:
    response = gateway_client.chat.completions.create(
        model=config.get('OpenAI', 'MODEL'),
        messages=[
            {"role": "system", "content": system_content},
            {"role": "user",   "content": debug_query},
        ],
        max_tokens=100,
        temperature=0,
    )
