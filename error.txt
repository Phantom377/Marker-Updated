fernet = Fernet(SAP_DATA_KEY.encode())

def decrypt_value(value: str) -> str:
    if not value:
        return ""
    return fernet.decrypt(value.encode()).decode()

def load_queries_by_file() -> list[tuple[str, list[dict]]]:
    """Returns a list of (filename_stem, queries) — one entry per input file."""
    files = []
    for fname in sorted(os.listdir(QUERIES_FOLDER)):
        if not fname.endswith(".json"):
            continue
        with open(QUERIES_FOLDER / fname, "r", encoding="utf-8") as f:
            data = json.load(f)

        queries = []
        if isinstance(data, dict) and "sections" in data:
            for section in data.get("sections", []):
                for qa in section.get("qa_pairs", []):
                    if "question" in qa:
                        queries.append({
                            "query": qa["question"],
                            "question_number": qa.get("question_number"),
                            "section": section.get("section", ""),
                            "policy_source": data.get("policy_source", ""),
                        })
        elif isinstance(data, list):
            queries.extend([{"query": q} for q in data])
        elif isinstance(data, dict) and "query" in data:
            queries.append({"query": data["query"]})

        if queries:
            files.append((Path(fname).stem, queries))
    return files

def fetch_user_data(required_sap_fields: list[str]) -> dict:
    payload = json.dumps({
        "query": USERNAME,
        "searchType": "vector",
        "vector": {
            "kind": "vector",
            "fields": "user_name_vector",
            "k": 3,
            "exhaustive": True,
        },
        "$select": "user_name, " + ", ".join(required_sap_fields),
        "top": 1
    })
    response = requests.post(SAP_URL, headers=SAP_HEADERS, data=payload)
    if response.status_code != 200:
        print(f"SAP search error {response.status_code}: {response.text}")
        return {}

    result = response.json().get("value", [])
    if not result:
        return {}

    raw = result[0]
    fields_to_decrypt = ["user_name"] + required_sap_fields
    return {f: decrypt_value(raw.get(f, "")) for f in fields_to_decrypt}

def debug_pipeline(query: str, system_content: str, az_openai_client, start_trace,
                   az_model: str, employee_data: dict = None):
    """
    Runs each step independently and reports which step triggers the PII error.
    Pass in a hardcoded employee_data dict to test step 4 in isolation.
    """
    print("\n=== DEBUG MODE ===")

    # Step 1 — Query + retrieval
    print("\n[Step 1] run_conversation_core (query + policy retrieval via Azure)...")
    try:
        events = list(llm_core.run_conversation_core(
            messages={"messages": [
                {"role": "system", "content": system_content},
                {"role": "user",   "content": query}
            ]},
            system_content=system_content,
            openai_client=az_openai_client,
            model_name=az_model,
        ))
        completion_text, retrieval_list = "", []
        for event in events:
            if event[0] == "chunk":
                completion_text += event[1]
            elif event[0] == "done":
                _, _, _, retrieval_list = event[1:]
        print(f"  ✓ OK — got {len(retrieval_list)} chunks, {len(completion_text)} chars")
    except Exception as e:
        print(f"  ✗ FAILED at Step 1: {e}")
        return

    # Step 2 — Identify SAP fields
    print("\n[Step 2] identify_required_personalization_fields (policy chunks only)...")
    try:
        required_sap_fields = llm_core.identify_required_personalization_fields(
            az_openai_client=az_openai_client,
            model_name="gpt-4o",
            retrieval_list=retrieval_list,
            start_trace=start_trace,
        )
        if isinstance(required_sap_fields, str):
            required_sap_fields = ast.literal_eval(required_sap_fields)
        print(f"  ✓ OK — required fields: {required_sap_fields}")
    except Exception as e:
        print(f"  ✗ FAILED at Step 2: {e}")
        return

    # Step 3 — SAP fetch (no LLM call, just API)
    print("\n[Step 3] fetch_user_data (SAP vector search, no LLM)...")
    try:
        if employee_data is None:
            employee_data = fetch_user_data(required_sap_fields) if required_sap_fields else {}
        print(f"  ✓ OK — employee fields returned: {list(employee_data.keys())}")
    except Exception as e:
        print(f"  ✗ FAILED at Step 3: {e}")
        return

    # Step 4 — Personalize (sends employee data to LLM)
    print("\n[Step 4] personalize_response (policy chunks + employee data → LLM)...")
    try:
        personalized_text = llm_core.personalize_response(
            az_openai_client=az_openai_client,
            model_name="gpt-4o",
            query=query,
            retrieval_list=retrieval_list,
            employee_data=employee_data,
            start_trace=start_trace,
        )
        print(f"  ✓ OK — {len(personalized_text)} chars")
    except Exception as e:
        print(f"  ✗ FAILED at Step 4: {e}")
        print("  → The PII is likely in the decrypted employee_data being sent to the LLM.")
        print(f"  → Employee data keys sent: {list(employee_data.keys())}")
        return

    print("\n=== All steps passed ===")
    return personalized_text


def run_pipeline(query: str, system_content: str, azure_config: dict, start_trace, az_model: str) -> str:
    # Create a fresh client per question so abandoned connections from timeouts
    # don't exhaust the shared connection pool
    az_openai_client = AzureOpenAI(
        api_key=azure_config['api_key'],
        api_version=azure_config['api_version'],
        azure_endpoint=azure_config['azure_endpoint'],
        timeout=TIMEOUT_SECONDS,
    )

    # Step 1 — Generate draft + retrieve chunks (via Azure OpenAI directly)
    events = list(llm_core.run_conversation_core(
        messages={"messages": [
            {"role": "system", "content": system_content},
            {"role": "user",   "content": query}
        ]},
        system_content=system_content,
        openai_client=az_openai_client,
        model_name=az_model,
    ))

    completion_text = ""
    retrieval_list  = []
    trace_metric_list = {}
    for event in events:
        if event[0] == "chunk":
            completion_text += event[1]
        elif event[0] == "done":
            _, trace_metric_list, doc_dict, retrieval_list = event[1:]

    # Step 2 — Identify required SAP fields from policy chunks
    required_sap_fields = llm_core.identify_required_personalization_fields(
        az_openai_client=az_openai_client,
        model_name=az_model,
        retrieval_list=retrieval_list,
        start_trace=start_trace,
    )
    if isinstance(required_sap_fields, str):
        required_sap_fields = ast.literal_eval(required_sap_fields)

    # Step 3 — Fetch + decrypt employee data from SAP
    employee_data = {}
    if required_sap_fields:
        employee_data = fetch_user_data(required_sap_fields)

    # Step 4 — Personalize response
    if employee_data:
        personalized_text = llm_core.personalize_response(
            az_openai_client=az_openai_client,
            model_name=az_model,
            query=query,
            retrieval_list=retrieval_list,
            employee_data=employee_data,
            start_trace=start_trace,
        )
    else:
        personalized_text = completion_text

    return personalized_text


def main():
    config = Config()
    os.environ["LANGFUSE_PUBLIC_KEY"] = config.get('tracing', 'LANGFUSE_PUBLIC_KEY')
    os.environ["LANGFUSE_SECRET_KEY"] = config.get('tracing', 'LANGFUSE_SECRET_KEY')
    os.environ["LANGFUSE_HOST"]       = config.get('tracing', 'LANGFUSE_HOST')

    lf = get_client()
    system_content = get_prompt_template('SYS_PROMPT')

    azure_config = {
        'api_key':        config.config['AZURE_OPENAI_API_KEY'],
        'api_version':    config.get('models', 'AZURE_OPENAI_API_VERSION'),
        'azure_endpoint': config.get('models', 'AZURE_OPENAI_ENDPOINT'),
    }
    az_model = config.get('models', 'AZURE_OPENAI_CHAT_DEPLOYMENT')

    files = load_queries_by_file()
    total_files = len(files)
    print(f"Loaded {total_files} input file(s)\n")

    timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")

    for file_idx, (fname_stem, queries) in enumerate(files, 1):
        print(f"=== File {file_idx}/{total_files}: {fname_stem}.json ({len(queries)} questions) ===\n")

        answered = []
        out_path = ANSWERED_FOLDER / f"{fname_stem}_{timestamp}.json"

        for item in queries:
            query = item["query"]
            print(f"[{item.get('policy_source','')} | {item.get('section','')} | Q{item.get('question_number','')}]")
            print(f"Q: {query}")

            error_message = ""
            try:
                with lf.start_as_current_observation(as_type="span", name="QnA") as start_trace:
                    executor = ThreadPoolExecutor(max_workers=1)
                    future = executor.submit(
                        run_pipeline, query, system_content, azure_config, start_trace, az_model
                    )
                    try:
                        personalized_text = future.result(timeout=TIMEOUT_SECONDS)
                    finally:
                        executor.shutdown(wait=False)

                    start_trace.update_trace(
                        input=query,
                        output=personalized_text,
                        user_id='pipeline',
                    )
                    lf.flush()

                print(f"A: {personalized_text}\n{'-'*60}\n")
                status = "answered"

            except FuturesTimeoutError:
                print(f"TIMEOUT after {TIMEOUT_SECONDS}s — skipping question.\n{'-'*60}\n")
                personalized_text = ""
                status = "timeout"
                error_message = f"TimeoutError: exceeded {TIMEOUT_SECONDS}s"

            except Exception as e:
                print(f"ERROR: {e} — skipping question.\n{'-'*60}\n")
                personalized_text = ""
                status = "error"
                error_message = f"{type(e).__name__}: {e}"

            answered.append({
                "policy_source":   item.get("policy_source", ""),
                "section":         item.get("section", ""),
                "question_number": item.get("question_number"),
                "question":        query,
                "answer":          personalized_text,
                "status":          status,
                "error":           error_message,
            })

            # Save incrementally after every question
            with open(out_path, "w", encoding="utf-8") as f:
                json.dump(answered, f, ensure_ascii=False, indent=2)
            print(f"Saved {len(answered)}/{len(queries)} → {out_path}")

        print(f"\nFile done. {sum(1 for a in answered if a['status'] == 'answered')} answered, "
              f"{sum(1 for a in answered if a['status'] == 'timeout')} timed out, "
              f"{sum(1 for a in answered if a['status'] == 'error')} errored.\n")


if __name__ == "__main__":
    main()
